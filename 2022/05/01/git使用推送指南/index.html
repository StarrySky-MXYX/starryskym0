GIT/git推送使用指南

git config –-global user.name “用户名” //需要引号哦
git config –-global user.email “邮箱” //需要引号哦
git init //第一次使用需要
git add . //添加所有文件，全局推送
git commit -m “推送信息”
git remote add origin 仓库地址 //写你的推送地址就好了，https或者git都可以
git push -u origin “master”（git push -f origin “master”强制推送）
git remote add origin 仓库地址 //增加推送源
git remote set-url origin 仓库地址 //删除并重新添加推送源
git remote remove origin //删除远程
